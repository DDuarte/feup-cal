154 void ByteBuffer::PutBits(size_t pos, T value, uint32 bitCount)
155 {
156  if (!bitCount)
157  throw ByteBufferException(/* (pos + bitCount) / 8, size(), 0 */);
158 
159  if (pos + bitCount > size() * 8)
160  throw ByteBufferException(/* false, (pos + bitCount) / 8, size(), (bitCount - 1) / 8 + 1 */);
161 
162  for (uint32 i = 0; i < bitCount; ++i)
163  {
164  size_t wp = (pos + i) / 8;
165  size_t bit = (pos + i) % 8;
166  if ((value >> (bitCount - i - 1)) & 1)
167  _buffer[wp] |= 1 << (7 - bit);
168  else
169  _buffer[wp] &= ~(1 << (7 - bit));
170  }
171 }
